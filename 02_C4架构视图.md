## C4 — 核心类图

```mermaid
classDiagram
    class CrossPlatformParser {
        +parseDouyinEvent(json: String): VideoEvent
        +parseBilibiliEvent(json: String): VideoEvent
        +parseXiaohongshuEvent(json: String): VideoEvent
        +parseWechatEvent(json: String): VideoEvent
        +parseYoutubeEvent(json: String): VideoEvent
        +parseWeiboEvent(json: String): VideoEvent
        +normalize(event: PlatformEvent): VideoEvent
    }
    
    class PlayAggregator {
        +aggregateTotal(events: DStream): Dataset
        +groupByPlatform(events: DStream): Dataset
        +trackNewVideo(events: DStream, videoId: String): Dataset
        +calcGrowthRate(events: DStream): Dataset
    }
    
    class HotspotDetector {
        +baselineFreq: Map[String, Double]
        +detectTrendingTopic(events: DStream): Dataset
        +calcBurstScore(topic: String, freq: Double): Double
        +crossPlatformCorrelation(topics: DStream): Dataset
    }
    
    class CompetitorTracker {
        +competitorIds: List[String]
        +trackPlays(events: DStream): Dataset
        +trackFollowers(events: DStream): Dataset
        +calcChangeRate(current: Double, previous: Double): Double
    }
    
    class TopicRecommender {
        +model: ALSModel
        +timeSeriesModel: ARIMAModel
        +recommend(creatorId: String, n: Int): List[TopicRecommendation]
        +predictTrend(topicType: String): TrendPrediction
    }
    
    class TimeRecommender {
        +clusterModel: KMeansModel
        +clusterActiveHours(followerData: DataFrame): DataFrame
        +recommend(creatorId: String): List[TimeSlot]
        +estimatePlayGain(timeSlot: TimeSlot): Double
    }
    
    CrossPlatformParser --> PlayAggregator : events
    CrossPlatformParser --> HotspotDetector : topic events
    CrossPlatformParser --> CompetitorTracker : competitor events
    PlayAggregator --> TopicRecommender : 播放趋势
    HotspotDetector --> TopicRecommender : 热点信号
    CompetitorTracker --> TopicRecommender : 竞品信号
```
